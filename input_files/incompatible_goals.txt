CONSTANTS:
    D_platoon       := 10
    RSSI_net        := 60
    ZERO            := 0
    MAX_STEERING    := 1

GOAL:
	NAME:
		accelerate_distance
    DESCRIPTION:
        accelerate when the distance from the car in front is more than the desired distance
    VARIABLES:
		distance_front      := REAL
		velocity_ego_t      := REAL
		velocity_ego_tnext  := REAL
	ASSUMPTIONS:
		distance_front > D_platoon
	GUARANTEES:
		velocity_ego_tnext > velocity_ego_t

GOAL:
	NAME:
		decelerate_distance
	DESCRIPTION:
        accelerate when the distance from the car in front is more than the desired distance
	VARIABLES:
		distance_front      := REAL
		velocity_ego_t      := REAL
		velocity_ego_tnext  := REAL
	ASSUMPTIONS:
		distance_front < D_platoon
	GUARANTEES:
		velocity_ego_tnext < velocity_ego_t

GOAL:
	NAME:
		maintainspeed_distance
	DESCRIPTION:
        maintain the speed when the distance from the car in front is equal to the desired distance
	VARIABLES:
		distance_front      := REAL
		velocity_ego_t      := REAL
		velocity_ego_tnext  := REAL
	ASSUMPTIONS:
		distance_front == D_platoon
	GUARANTEES:
		velocity_ego_tnext == velocity_ego_t

ENDGOALS
